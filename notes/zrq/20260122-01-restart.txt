# Create our build and dist directories.
#[root@action-runner]

    #
    # Set our target version.
    baseversion=0.0.2
    javaversion=${baseversion:?}-SNAPSHOT-$(date '+%Y%m%d')

    #
    # Update a gitignore file.
    checkignore()
        {
        local target=${1:?}
        local targetpath=$(dirname  ${target})
        local targetname=$(basename ${target})
        local targetfile=${targetpath}/.gitignore

        if [ ! -e "${targetfile}" ]
        then
            touch "${targetfile}"
        fi

        grep -qxF "${targetname}" "${targetfile}"
        if [ $? -ne 0 ]
        then
            echo "${targetname}" >> "${targetfile:?}"
        fi
        }

    #
    # Check a directory exists.
    checkexists()
        {
        local target=${1:?}

        if [ ! -e "${target}" ]
        then
            mkdir \
                --parents \
                "${target}"
        fi
        }

    #
    # Create our schema build directory.
    checkignore \
        /trebula/build

    checkexists \
        /trebula/build
